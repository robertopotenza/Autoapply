    <script>
        // Check authentication
        const token = localStorage.getItem('authToken');
        if (!token) {
            window.location.href = '/login.html';
        }

        // Display user email
        const userEmail = localStorage.getItem('userEmail');
        if (userEmail) {
            document.getElementById('user-email').textContent = userEmail;
        }

        // Load dashboard data
        async function loadDashboard() {
            try {
                const response = await fetch('/api/wizard/completion', {
                    headers: {
                        'Authorization': `Bearer ${token}`
                    }
                });

                if (response.status === 401 || response.status === 403) {
                    logout();
                    return;
                }

                const data = await response.json();

                if (data.success) {
                    const { percentage, sections } = data.data;

                    // Update completion percentage
                    document.getElementById('completion-percentage').textContent = `${percentage}%`;
                    document.getElementById('completion-bar-fill').style.width = `${percentage}%`;

                    // Update section statuses
                    updateSectionStatus('step1', sections.jobPreferences);
                    updateSectionStatus('step2', sections.jobPreferences); // Same section in DB
                    updateSectionStatus('step3', sections.profile);
                    updateSectionStatus('step4', sections.eligibility);
                }
            } catch (error) {
                console.error('Error loading dashboard:', error);
            }
        }

        function updateSectionStatus(step, isComplete) {
            const statusEl = document.getElementById(`status-${step}`);
            if (isComplete) {
                statusEl.textContent = 'Complete';
                statusEl.classList.remove('status-incomplete');
                statusEl.classList.add('status-complete');
            }
        }

        function editSection(step) {
            // Store which step to open
            localStorage.setItem('wizardStep', step);
            window.location.href = '/wizard.html';
        }

        function startApplying() {
            fetch('/api/wizard/completion', {
                headers: {
                    'Authorization': `Bearer ${token}`
                }
            })
            .then(res => res.json())
            .then(data => {
                if (data.data.percentage < 100) {
                    alert('Please complete your profile before starting auto-apply');
                    return;
                }
                alert('Auto-apply feature will be activated!');
                // TODO: Activate auto-apply
            });
        }

        function logout() {
            localStorage.removeItem('authToken');
            localStorage.removeItem('userEmail');
            localStorage.removeItem('userId');
            window.location.href = '/login.html';
        }

        // Load dashboard on page load
        loadDashboard();
    </script>
